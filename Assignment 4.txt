    <title>Basic Math</title>
    <style>
        /* All tables */
        table {
            border-collapse: collapse;
        }

        /* All table cells */
        td {
            border: solid 1px silver;
            text-align: right;
        }

        /* All textboxes with type=number */
        input[type=number] {
            width: 80px;
            text-align: right;
        }

        /* Results table only */
        #results {
            margin-top: 2em;
            /* Invisible when page first opens */
            display: none;
        }
    </style>
    <script>
        function calculate() {
            var quantity = document.getElementById("tbQuantity").value;
            var unitprice = document.getElementById("tbUnitPrice").value;
            // Sales tax percent is textbox value divided by 100
            var salestaxpercent = document.getElementById("tbSalesTaxPercent").value / 100;
            // Do the math
            var extendedprice = quantity * unitprice;
            var salestaxamount = extendedprice * salestaxpercent;
            var grandtotal = extendedprice + salestaxamount;
            // Populate the results table cells.
            document.getElementById("tdTotalSale").innerHTML = "$" + extendedprice.toFixed(2);
            document.getElementById("tdSalesTax").innerHTML = "$" +salestaxamount.toFixed(2);
            document.getElementById("tdGrandTotal").innerHTML = "$" +grandtotal.toFixed(2);
            //Make the results table visble (CSS made in invisible initially)
            document.getElementById("results").style.display="block";
        }
    </script>
</head>

<body>
    <table style="text-align:right;">

            <td><input type="number" id="tbSalesTaxPercent" value="5.75"></td>

            <td colspan="2" style="text-align:center;">
                <button onclick="calculate()">Calculate</button>
